apiVersion: networking.k8s.io/v1beta1

  name: unifi-protect
  annotations:
    kubernetes.io/ingress.class: internal
    nginx.ingress.kubernetes.io/backend-protocol: "HTTPS"

  name: unifi-protect

  ports:
